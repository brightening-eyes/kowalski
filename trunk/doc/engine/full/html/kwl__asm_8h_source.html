00188 static inline void kwlApplyGainRamp(float* outBuffer, 
00189                                     int numOutChannels, 
00190                                     int numFrames, 
00191                                     float startGain[2], 
00192                                     float endGain[2])
00193 {
00194     float deltaGain[2] = 
00195     {
00196         endGain[0] - startGain[0],
00197         endGain[1] - startGain[1]
00198     };
00199     
00200     const int numSamples = numOutChannels * numFrames;
00201     
00202     /* 
00203      If the gain difference between consecutive frames is less than this,
00204      a gain ramp will not be applied.
00205      */
00206     const float eps = 1e-7f;
00207     
00208     for (int ch = 0; ch < numOutChannels; ch++)
00209     {
00210         float gain = startGain[ch];
00211         float deltaGainPerFrame = deltaGain[ch] / numFrames;
00212         
00213         if (deltaGainPerFrame < eps && deltaGainPerFrame > -eps)
00214         {
00215             /* Gain difference too small, don't apply ramp */
00216             for (int i = ch; i < numSamples; i += numOutChannels)
00217             {
00218                 outBuffer[i] *= gain;
00219             }
00220         }
00221         else 
00222         {
00223             /* Apply gain ramp */
00224             for (int i = ch; i < numSamples; i += numOutChannels)
00225             {
00226                 outBuffer[i] *= gain;
00227                 gain += deltaGainPerFrame;
00228             }
00229         }
00230     }
00231 }
