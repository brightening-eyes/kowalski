00085 static inline int decodeNibble(int nibble, int *predictor, int*stepIndex)
00086 {
00087     int p = *predictor;
00088     int s = *stepIndex;
00089     
00090     /*compute a delta to add to the predictor value*/
00091     int diff = 0;
00092     if (nibble & 4)
00093     {
00094         diff += KWL_IMA_ADPCM_STEP_TABLE[s];
00095     }
00096     if (nibble & 2)
00097     {
00098         diff += KWL_IMA_ADPCM_STEP_TABLE[s] >> 1;
00099     }
00100     if (nibble & 1)
00101     {
00102         diff += KWL_IMA_ADPCM_STEP_TABLE[s] >> 2;
00103     }
00104     diff += KWL_IMA_ADPCM_STEP_TABLE[s] >> 3;
00105     
00106     if (nibble & 8)
00107     {
00108         diff = -diff;
00109     }
00110     
00111     /*add the delta and clamp the predictor*/
00112     p += diff;
00113     if (p > 32767)
00114     {   
00115         p = 32767;
00116     }
00117     else if (p < -32768) 
00118     {
00119         p = -32768; 
00120     }
00121     
00122     /*compute a new step index*/
00123     s += KWL_IMA_ADPCM_INDEX_TABLE[nibble];
00124     if (s > 88) 
00125     {
00126         s = 88;
00127     }
00128     else if (s < 0) 
00129     {
00130         s = 0;
00131     }
00132     
00133     *predictor = p;
00134     *stepIndex = s;
00135
